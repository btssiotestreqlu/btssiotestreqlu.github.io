import base64


def makepass():
  password = "utec"
  flag = base64.b64encode(password.encode()).decode()
  check_flag(input("Entrez le mot de passe : "),flag)

def check_flag(args,flags):
   if args == flags:
     print("Mot de passe correct !")
   else:
     print("Mot de passe incorrect !")

if __name__=="__main__":
   makepass()
